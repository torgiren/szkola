select emp.ename, dept.dname from emp left join dept on emp.deptno=dept.deptno;
select emp.ename as workers_in_dallas from emp left join dept on emp.deptno=dept.deptno where dept.loc='DALLAS';
select emp.ename, emp.sal,salgrade.grade from emp,salgrade where emp.sal between salgrade.losal and salgrade.hisal order by emp.sal;
select emp.ename, emp.sal,salgrade.grade from emp,salgrade where emp.sal between salgrade.losal and salgrade.hisal and salgrade.grade=3 order by emp.sal;
select prac.ename, prac.sal, kier.sal from emp as prac, emp as kier where prac.sal<kier.sal and prac.mgr=kier.empno;
select emp.job from emp where emp.deptno=10 or emp.deptno=30 group by job;
select avg(emp.sal) as sredni_zarobek from emp;
select min(sal) as min_dla_clerk from emp where job='CLERK';
select count(emp.empno) as pracownikow_w_dept_20 from emp where emp.deptno=20;
select emp.job, avg(emp.sal) from emp group by emp.job;
select emp.job, max(emp.sal) from emp group by emp.job;
select dept.dname, avg(emp.sal)  from dept,emp where dept.deptno=emp.deptno group by dept.dname having count(emp.empno)>3;
select max(emp.sal)-min(emp.sal) as roznica from emp;
 select kier.ename, count(prac.empno) from emp as kier, emp as prac where prac.mgr=kier.empno group by kier.ename;
 select salgrade.grade,avg(emp.sal) from emp,salgrade where emp.sal between salgrade.losal and salgrade.hisal group by grade order by grade;
select salgrade.grade,count(emp.empno) from emp,salgrade where emp.sal between salgrade.losal and salgrade.hisal group by grade order by grade;
select emp.ename from emp where emp.sal=(select min(sal) from emp);
select dept.dname as dzial from dept join emp on dept.deptno=emp.deptno where emp.job='CLERK' group by dzial;
select prac.ename from emp as prac where prac.sal>(select avg(emp.sal) from emp where emp.deptno=prac.deptno);
